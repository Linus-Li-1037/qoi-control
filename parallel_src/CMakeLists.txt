find_package(MPI REQUIRED)

function(add_my_executable_prl target source)
    add_executable(${target} ${source})
    target_include_directories(${target} PRIVATE ${MDR_INCLUDES} ${MGARDx_INCLUDES} ${SZ3_INCLUDES} ${ZSTD_INCLUDES})
    target_link_libraries(${target} ${PROJECT_NAME} adios2::adios2 ${SZ3_LIB} ${ZSTD_LIB})
endfunction()

add_my_executable_prl(refactorGE refactorGE.cpp)
